  let currentDoc = changeEvent.fullDocument;
  let responseJson = null;
  
  if(typeof currentDoc === 'undefined') {
    currentDoc = {}
  }
  
  try {
    const body = { "name": currentDoc.name, "duration": currentDoc.duration, "movie": currentDoc.movie };
    
    const response = await context.http.post({
      url: "<YOUR WEBHOOK URL ENDPOINT>",
      body: JSON.stringify(body)
    });
    
    responseJson = EJSON.parse(response.body.text());  
  } catch(e) {
    return e; 
  }
  
  return responseJson;
};
