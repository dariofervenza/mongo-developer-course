function recs_older_than(days) {
  days_in_millis = days*24*60*60*1000
  today= new Date()
  n_days_ago =  new Date(today - days_in_millis)
  return  { create_date: { "$lte" : n_days_ago}}
}
pipe = [{$match : recs_older_than(180)}]
db.people.aggregate(pipe)

totalstart = new Date()
myarray = []
myarray = new Array(1440).fill(0) 
db.events.deleteOne({_id:"test"})
for(day=0;day<50;day++) {
    print("Day: "+day)
    db.events.updateOne({_id:"test"},{$push: {a: myarray} }, {upsert:true})
    eventtimes = []
    for(x=0;x<2000;x++)  { eventtimes.push(Math.floor(Math.random()*1440)) }
    bulk = db.events.initializeUnorderedBulkOp()
    for(x=0;x<2000;x++) {
        inc = {}
        d = eventtimes[x]
        inc[`a.${day}.${d}`] = 1
        bulk.find({_id:"test"}).updateOne({$inc:inc})
    }
    start = new Date()
    bulk.execute()
    time = new Date() - start
    docsize = bsonsize(db.events.findOne({_id:"test"}))
    print("Update time: " + time + "ms  / Document size: " + (docsize/1024).toFixed(1) + "KB")
}
totaltime = new Date() - totalstart
print("Total time: " + totaltime + "ms")
